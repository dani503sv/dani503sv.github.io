//clases POO

class Animal{

 //constructor de la clase Animal
 //teniendo parametros nombre y genero
 //Inicializamos las propiedades de la clase

 constructor(nombre, genero){
  this.nombre = nombre;
  this.genero = genero;
 }

 //metodo
 sonido(){
  console.log("El animal hace un sonido");
 }

 saludar(){
  console.log("Hola soy "+this.nombre+" y soy un "+this.genero);
 }
}

//crear objecto de la clase Animal
const sasha = new Animal("Sasha", "Hembra"),
shapo = new Animal("Shapo", "Macho");

//ver objectos en consola
console.log(sasha);
console.log(shapo);
shapo.saludar();
shapo.sonido(); //llama al metodo sonido de la clase Animal
sasha.saludar();

HERENCIA

//HERENCIA DE CLASES
//usando extends para heredar de la clase Animal

class Perro extends Animal{
 constructor(nombre, genero, tamanio){
  //nombre y genero son parametros de la clase Animal
  //no usar this en super, ya que no se ha inicializado
  super(nombre, genero); //llamamos al constructor de la clase Animal
  this.tamanio = tamanio; //agregamos una nueva propiedad tamanio
 }

 //sobreescribimos el metodo sonido de la clase Animal
 sonido(){
  console.log("Soy un perro y ladro");
 }

 ladrar(){
  console.log(" Guau guau, soy "+this.nombre+" y soy un "+this.tamanio);
 }

}

//crear objecto de la clase Perro
const blaki = new Perro("Blaki", "Hembra", "Pequeña");
console.log(blaki);
blaki.saludar();
blaki.ladrar();
Metodos GET Y SET

GET Y SET

//METODO GET Y SET
class Personas{
 constructor(){
   this.nombre = null; // propiedad privada
   this.edad = null; // propiedad privada
 }

 //SE USAN COMO PROPIEDADES
 //metodo get para obtener el nombre
 get getNombre(){
   return this.nombre;
  }

 //metodo set para establecer el nombre
 set setNombre(nombre){
   this.nombre = nombre;
  }

 //metodo get para obtener la edad
 get getEdad(){
   return this.edad;
 }

 //metodo set para establecer la edad
 set setEdad(edad){
   this.edad = edad;
 }

 //OBJECTO ESTATiCO
 static metodoEstatico(){
   console.log("Esta es una clase de ejemplo para demostrar el uso de métodos static.");
 }

}

// Crear una instancia de la clase Personas
const persona = new Personas();

// Establecer valores usando los métodos set
persona.setEdad = "Daniel";
persona.setNombre = 20;

// Obtener valores usando los métodos get
console.log("Nombre: " + persona.getNombre); // Imprime: Nombre: Daniel
console.log("Edad: " + persona.getEdad); // Imprime: Edad: 20

//llamar al metodo estatico
Personas.metodoEstatico(); // Imprime: Esta es una clase de ejemplo para demostrar el uso de métodos static.

Objecto Date

Date

//DATE
let fecha = new Date();
console.log(fecha);

//Dia del mes
console.log("Day: " +fecha.getDate());
console.log("Day de la semana [0-6]: " +fecha.getDay()); //Dia de la semana (0-6)

//Mes
console.log("Mes: " + fecha.getMonth()); //Mes (0-11)

//año
console.log("Año: " + fecha.getFullYear()); //Año

//Hora, minutos, segundos
console.log("Hora: " + fecha.getHours()); //Hora
console.log("Minutos: " + fecha.getMinutes()); //Minutos
console.log("Segundos: " + fecha.getSeconds()); //Segundos

//Usando toString
console.log(fecha.toDateString()); //Fecha en formato legible
console.log(fecha.toLocaleString()); //Fecha y hora en formato local
console.log(fecha.toLocaleTimeString()); //Hora en formato local
