Function(){}

Forma de usar diferentes Fuction

Metodo SET
//========METODO SET==========
function metodo()
{
console.log("esta es 1 funcion SET")
console.log("esta es 2 funcion SET")
console.log("esta es 3 funcion SET")
}
//mandar a llamar la funcion
metodo();
Metodo GET
//==========METODO GET==========
function metodoGet()
{
let cadena = "Esta es la funcion GET"
return cadena;
}
//GET se llama dentro de console.log()
console.log(metodoGet());
Metodo Parametro
//==========METODO PARAMETRO==========
let edad = 18;
let nombre = "daniel";
function metodoParametro(edad, nombre)
{
console.log(`Nombre: ${edad} y edad: ${nombre}`);
}
//llamar metodo
metodoParametro(edad, nombre);
Metodo Anonimo
//==========FUNCION ANONIMA==========
const funcionExpresada = function(){
console.log("Funcion asignada a una variable, no se puede poner antes de su creacion");
}
funcionExpresada();
Tip: Usar nombres descriptivos y parámetros claros.
Arrays []

Forma de usar Arreglos

[] Es un arreglo o array SIEMPRE inicia de 0 +
// 0 1 2
const arregloA = [1, true, "Un arreglo"];

//Llamar al arreglo
console.log(arregloA);

//Llamar al arreglo en una posicion
console.log(arregloA[2]);

//inicializar arreglo y asignar el mismo valor
const d = Array(5).fill(2);

//Llamar al arreglo
console.log(d);
Metodo push("X")
//metodo push("Nuevo") para agregar un nuevo valor al final
arregloA.push("Nuevo");
console.log(arregloA);
Metodo pop()
//metodo pop() para quitar ultimo valor
arregloA.pop();
console.log(arregloA);
Metodo forEach()
//metodo forEach recorrer cada valor
// indice el numero de array 0+
// elemento que guarda la array
const letra = ["A", "B", "C"];
letra.forEach(
function (elemento, indice){
console.log("li id="${indice}"> ${elemento} /li");
}
);
Tip: Usa métodos de array para facilitar tu trabajo JavaScript tiene métodos muy útiles para trabajar con arreglos. Uno de los más importantes para principiantes es .forEach()..
