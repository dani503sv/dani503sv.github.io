//Se usa em framework
//...c para hacerlo infinito

function sumar(a,b, ...c){
 let resultado = a+b;
 //sumar los valores infinitos que recibira C
  c.forEach(
   function (n){
   resultado += n;

//CLASE PADRE

function Persona(nombre, edad, tel){
 this.nombre = nombre;
 this.edad = edad;
 this.tel = tel;
}

//CREAMOS UN PROTIPO PARA UN METODO
Persona.prototype.imprimir = function(){
 console.log("nombre: "+this.nombre+ " edad: "+this.edad
   + " tel: "+ this.tel
 );
}

//CREAMOS UN OBJECTO Y LE DAMOS SUS PARAMETROS
const personaX = new Persona("D", "21", "2121");

//llamar al metodo por prototipo Y VER COMO FUNCIONA
personaX.imprimir();

//HERENCIA
//CLASE HIJA
//DONDE USAMOS this.super para heredes constructor y para pasarle argumentos

function Usuario(usuario, pass, nombre, edad, tel){
 //heredar Persona como hijo
  this.super = Persona;
 this.super(nombre, edad, tel);
 this.usuario = usuario;
 this.pass = pass;
}

//sacar instancia Usuario hereda de persona
Usuario.prototype = new Persona();
Usuario.prototype.constructor = Usuario;

//sobreescribir metodos prototipo
Usuario.prototype.imprimir = function(){
 console.log("NOMBRE: "+this.nombre+ " EDAD: "+this.edad
  + " TEL: "+ this.tel
 );
};

Usuario.prototype.usuarioImprimir = function(){
 console.log("usuario: "+this.usuario + " pass: "+this.pass);
 this.imprimir();
}
const user = new Usuario("usuario", "pass", "Dani", 21, "2121");
user.usuarioImprimir();
